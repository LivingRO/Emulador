	case 3:

			mes "[Guardião do Portal]";

			mes "Que Deus o abençoe ..";
			close2;
			delitem .@item_req,1;
			warp getarg(2),getarg(3),getarg(4);
			end;
		} 
		else {
			mes "[Guardião do Portal]";
			mes ". . . . . .";
			mes "Infelizmente você não tem um privilégio. Para entrar nesta porta... Você não será capaz de passar pelos Alquimistas Ancestrais.";
			mes "Não vou deixar você ir.";
			close;
		}
	case 4:
		mes "[Guardião do Portal]";
		mes "Esta Torre do Relógio é o lugar onde três Lendários Alquimistas treinavam suas habilidades espirituais.";

		close;
	}
